// Toggle section open/close with accessibility
function toggleSection(sectionId) {
    const section = document.getElementById(sectionId);
    const header = section.querySelector('.ltc-section-header');
    const isExpanded = header.getAttribute('aria-expanded') === 'true';
    
    // Close all other sections
    document.querySelectorAll('.ltc-section').forEach(s => {
        if (s.id !== sectionId) {
            s.classList.remove('active');
            s.querySelector('.ltc-section-header').setAttribute('aria-expanded', 'false');
        }
    });
    
    // Toggle current section
    section.classList.toggle('active');
    header.setAttribute('aria-expanded', !isExpanded);
}

// Document ready function
document.addEventListener('DOMContentLoaded', function() {
    // Keyboard navigation for sections
    document.querySelectorAll('.ltc-section-header').forEach(header => {
        header.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' || e.key === ' ') {
                e.preventDefault();
                header.click();
            }
        });
    });

    // Smooth scroll for navigation links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
            e.preventDefault();
            const target = document.querySelector(this.getAttribute('href'));
            if (target) {
                target.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
                
                // If it's a section link, open the section
                if (target.classList.contains('ltc-section')) {
                    target.classList.add('active');
                    target.querySelector('.ltc-section-header').setAttribute('aria-expanded', 'true');
                }
            }
        });
    });

    // Add print functionality
    window.addEventListener('beforeprint', () => {
        // Expand all sections for printing
        document.querySelectorAll('.ltc-section').forEach(section => {
            section.classList.add('active');
        });
    });

    window.addEventListener('afterprint', () => {
        // Restore original state after printing
        document.querySelectorAll('.ltc-section').forEach(section => {
            section.classList.remove('active');
        });
    });

    // Intersection Observer for animations
    const observerOptions = {
        threshold: 0.1,
        rootMargin: '0px 0px -50px 0px'
    };

    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.classList.add('animated');
            }
        });
    }, observerOptions);

    // Observe all animated elements
    const animatedElements = document.querySelectorAll('.slide-up, .fade-in');
    animatedElements.forEach(el => {
        observer.observe(el);
    });

                    <div class="ltc-section-intro">
                        <h3>Why This Matters</h3>
                        <p>About 25% of people let their long-term care policies lapse before ever using them, wasting years of premium payments. Understanding these four key features ensures you can access your benefits when needed and maximize your coverage value.</p>
                    </div>
                    
                    <ul class="ltc-checklist">
                        <li class="ltc-checklist-item">
                            <div class="ltc-checklist-title">Benefit Triggers</div>
                            <div class="ltc-checklist-desc">
                                <p>Your policy pays when you meet one of these conditions:</p>
                                <ul>
                                    <li>Need hands-on help with 2+ Activities of Daily Living (ADLs): bathing, dressing, eating, toileting, transferring (moving), or continence</li>
                                    <li>Have severe cognitive impairment requiring substantial supervision for your safety</li>
                                </ul>
                                <p><strong>Action:</strong> Review your policy's specific trigger definitions - some are more generous than others.</p>
                            </div>
                        </li>
                        <li class="ltc-checklist-item">
                            <div class="ltc-checklist-title">Elimination Period</div>
                            <div class="ltc-checklist-desc">
                                <p>Your "time deductible" before benefits begin:</p>
                                <ul>
                                    <li>Common periods: 30, 60, or 90 days</li>
                                    <li>You pay all costs during this waiting period</li>
                                    <li>Some policies count calendar days, others count only days you receive care</li>
                                </ul>
                                <p><strong>Action:</strong> Know your elimination period and have funds set aside to cover this gap.</p>
                            </div>
                        </li>
                        <li class="ltc-checklist-item">
                            <div class="ltc-checklist-title">Inflation Protection</div>
                            <div class="ltc-checklist-desc">
                                <p>Protects against rising care costs:</p>
                                <ul>
                                    <li>Look for 3-5% annual compound growth</li>
                                    <li>Colorado care costs increase ~4-6% annually</li>
                                    <li>Without inflation protection, your coverage loses value each year</li>
                                </ul>
                                <p><strong>Action:</strong> Verify your inflation protection type and rate - simple interest is less valuable than compound.</p>
                            </div>
                        </li>
                        <li class="ltc-checklist-item">
                            <div class="ltc-checklist-title">Partnership Status</div>
                            <div class="ltc-checklist-desc">
                                <p>Colorado's Partnership program offers unique asset protection:</p>
                                <ul>
                                    <li>Every dollar your policy pays = one dollar of assets protected from Medicaid</li>
                                    <li>Only qualifying Partnership policies offer this protection</li>
                                    <li>Protection applies even if you move to another Partnership state</li>
                                </ul>
                                <p><strong>Action:</strong> Check if your policy has Partnership certification - look for "Partnership Qualified" language.</p>
                            </div>
                        </li>
                    </ul>

                    <div class="ltc-tip">
                        <strong>Premium Payment Tip:</strong>
                        Most policies waive premiums 60-90 days after you start receiving benefits. Mark this date when filing a claim to avoid overpaying!

                    <h2 class="ltc-section-title" id="section4-title">Critical Statistics: Why LTC Benefits Often Go Unused</h2>

                    <div class="ltc-section-intro">
                        <h3>Learn from the Data</h3>
                        <p>Understanding these statistics helps you avoid common pitfalls and ensures you'll actually use the coverage you've been paying for. Each statistic below includes actionable advice to help you beat the odds.</p>

                    <div class="ltc-facts-grid">
                        <div class="ltc-fact-card">
                            <span class="ltc-fact-number">25%</span>
                            <p class="ltc-fact-text">of policyholders let coverage lapse before using benefits</p>
                            <div class="ltc-fact-tip">
                                <span>💡</span>
                                <span>Set up autopay & review annually</span>
                            </div>
                        </div>
                        <div class="ltc-fact-card">
                            <span class="ltc-fact-number">30-50%</span>
                            <p class="ltc-fact-text">of new applicants age 60+ are declined coverage</p>
                            <div class="ltc-fact-tip">
                                <span>💡</span>
                                <span>You're fortunate to have coverage - keep it!</span>
                            </div>
                        </div>
                        <div class="ltc-fact-card">
                            <span class="ltc-fact-number">70%</span>
                            <p class="ltc-fact-text">of adults 65+ will need some form of long-term care</p>
                            <div class="ltc-fact-tip">
                                <span>💡</span>
                                <span>High probability you'll use your benefits</span>
                            </div>
                        </div>
                        <div class="ltc-fact-card">
                            <span class="ltc-fact-number">87%</span>
                            <p class="ltc-fact-text">of claims don't exhaust the full policy benefits</p>
                            <div class="ltc-fact-tip">
                                <span>💡</span>
                                <span>File early without fear of "running out"</span>
                            </div>
                        </div>
                        <div class="ltc-fact-card">
                            <span class="ltc-fact-number">3.9 years</span>
                            <p class="ltc-fact-text">average duration of long-term care needs for women</p>
                            <div class="ltc-fact-tip">
                                <span>💡</span>
                                <span>Ensure adequate benefit period</span>
                            </div>
                        </div>
                        <div class="ltc-fact-card">
                            <span class="ltc-fact-number">$4,500+</span>
                            <p class="ltc-fact-text">monthly cost of assisted living in Colorado (2024)</p>
                            <div class="ltc-fact-tip">
                                <span>💡</span>
                                <span>Inflation protection is essential</span>
                            </div>
                        </div>

                        <h3>Common Reasons Benefits Go Unused</h3>
                        <ul style="margin-left: 1.5rem; color: var(--text-secondary); line-height: 1.8;">
                            <li><strong>Lack of awareness:</strong> Family doesn't know about the policy</li>
                            <li><strong>Lost paperwork:</strong> Can't find policy details when needed</li>
                            <li><strong>Denial:</strong> Reluctance to admit care is needed</li>
                            <li><strong>Complexity:</strong> Claims process seems too difficult</li>
                            <li><strong>Premium lapses:</strong> Coverage canceled due to non-payment</li>
