+import { createContext, useCallback, useContext, useMemo, useState, type ReactNode } from "react";
+import ScheduleTourDialog from "@/components/ScheduleTourDialog";
+
+export interface ScheduleTourOptions {
+  communityId?: string;
+  communityName?: string;
+  title?: string;
+  description?: string;
+  urgencyText?: string;
+}
+
+interface ScheduleTourContextValue {
+  openScheduleTour: (options?: ScheduleTourOptions) => void;
+}
+
+const DEFAULT_OPTIONS: ScheduleTourOptions = {};
+
+const ScheduleTourContext = createContext<ScheduleTourContextValue | undefined>(undefined);
+
+export function ScheduleTourProvider({ children }: { children: ReactNode }) {
+  const [isOpen, setIsOpen] = useState(false);
+  const [options, setOptions] = useState<ScheduleTourOptions>(DEFAULT_OPTIONS);
+
+  const openScheduleTour = useCallback((overrides: ScheduleTourOptions = {}) => {
+    setOptions({ ...DEFAULT_OPTIONS, ...overrides });
+    setIsOpen(true);
+  }, []);
+
+  const handleOpenChange = useCallback((open: boolean) => {
+    setIsOpen(open);
+    if (!open) {
+      setOptions(DEFAULT_OPTIONS);
+    }
+  }, []);
+
+  const value = useMemo(() => ({ openScheduleTour }), [openScheduleTour]);
+
+  return (
+    <ScheduleTourContext.Provider value={value}>
+      {children}
+      <ScheduleTourDialog open={isOpen} onOpenChange={handleOpenChange} options={options} />
+    </ScheduleTourContext.Provider>
+  );
+}
+
+export function useScheduleTour() {
+  const context = useContext(ScheduleTourContext);
+  if (!context) {
+    throw new Error("useScheduleTour must be used within a ScheduleTourProvider");
+  }
+  return context;
